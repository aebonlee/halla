PDF 적재 → 청크

폴더의 PDF들을 읽어 청크로 나눕니다.

from langchain_community.document_loaders import PyPDFLoader
from langchain_text_splitters import CharacterTextSplitter
import os

splitter = CharacterTextSplitter(separator="\n", chunk_size=1000, chunk_overlap=50)
texts = []
for f in os.listdir("./pdfs"):
    if f.endswith(".pdf"):
        raw = PyPDFLoader(f"./pdfs/{f}").load()
        texts.extend(splitter.split_documents(raw))
print("총 청크 수:", len(texts))

벡터스토어 + retriever

from langchain_openai import OpenAIEmbeddings
from langchain_chroma import Chroma
db = Chroma.from_documents(texts, OpenAIEmbeddings())
retriever = db.as_retriever(search_kwargs={"k": 5})

RAG 체인 완성

검색한 근거(context)와 질문을 함께 넣어 답을 만듭니다 — 오픈북 시험처럼.

from langchain_core.prompts import ChatPromptTemplate
from langchain_core.output_parsers import StrOutputParser
from langchain_core.runnables import RunnablePassthrough
from langchain_openai import ChatOpenAI

template = ("너는 훌륭한 비서다. 아래 [내용]만 참고해 한국어로 답하라.\n"
            "[내용]\n{context}\n\n[질문]\n{question}")
prompt = ChatPromptTemplate.from_template(template)
def format_docs(ds): return "\n\n".join(d.page_content for d in ds)

chain = ({"context": retriever | format_docs, "question": RunnablePassthrough()}
         | prompt | ChatOpenAI(model="gpt-4o", temperature=0.1) | StrOutputParser())
print(chain.invoke("이 문서의 핵심은?"))

근거(metadata) 표시

"모르면 모른다"는 가드레일 + 출처(metadata)를 함께 보여주면 신뢰도가 올라갑니다.

def format_docs(ds):
    return "\n\n".join(f"내용:\n{d.page_content}\n메타데이터:\n{d.metadata}" for d in ds)

대화형으로 묶기

def ask(q): return chain.invoke(q)
while True:
    q = input("user >> ")
    if q.lower() == "quit": break
    print("ai >>", ask(q))

배포 — Gradio

완성한 체인에 채팅 화면을 붙여 공유 가능한 URL을 만듭니다.

import gradio as gr
from langchain_openai import ChatOpenAI
from langchain_core.messages import SystemMessage, HumanMessage, AIMessage
llm = ChatOpenAI(model="gpt-4o-mini", temperature=0.7)

def respond(message, history):
    msgs = [SystemMessage(content="너는 친절한 한국어 AI assistant야.")]
    for u, a in history:
        msgs += [HumanMessage(content=u), AIMessage(content=a)]
    msgs.append(HumanMessage(content=message))
    return llm.invoke(msgs).content

gr.ChatInterface(respond).launch(share=True)   # 공유 URL 생성

LangServe / LangGraph

체인을 REST API로 공개하려면 LangServe, 더 복잡한 흐름·상태가 필요하면 LangGraph로 확장합니다 (adv_02 노트북).
