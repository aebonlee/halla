0. 공통 준비 (한 번만)

아래 레시피들은 모두 이 준비를 마쳤다고 가정합니다. Colab 새 노트북에서 가장 먼저 실행하세요.

# 필요한 패키지 설치 (맨 앞 ! 는 "터미널 명령" 이라는 뜻)
!pip install langchain langchain-openai langchain-community langchain-chroma pypdf

import os
from google.colab import userdata                 # Colab '보안 비밀' 읽기
os.environ["OPENAI_API_KEY"] = userdata.get("OPENAI_API_KEY")  # 키를 환경변수로

from langchain_openai import ChatOpenAI
from langchain_core.prompts import ChatPromptTemplate
from langchain_core.output_parsers import StrOutputParser

# 거의 모든 레시피가 쓰는 기본 모델 (빠르고 저렴)
llm = ChatOpenAI(model="gpt-4o-mini", temperature=0)
⚠️ 키 보안
API 키는 코드에 직접 쓰지 말고 Colab 🔑(보안 비밀)에 넣어 불러옵니다. 노출되면 타인이 내 요금으로 씁니다 — "키 = 돈".

1. 한 줄 요약봇

긴 글을 넣으면 한 문장으로 줄여 줍니다. 프롬프트 → 모델 → 문자열의 가장 기본 체인.

# {text} 빈칸 하나를 가진 프롬프트
prompt = ChatPromptTemplate.from_template("다음 글을 한국어 한 문장으로 요약해줘:\n\n{text}")

# | (파이프)로 연결: 프롬프트 → 모델 → 문자열만 뽑기
chain = prompt | llm | StrOutputParser()

글 = "랭체인은 여러 AI 부품을 파이프로 이어 하나의 흐름을 만드는 도구다. ..."
print(chain.invoke({"text": 글}))   # invoke 에는 빈칸 값을 딕셔너리로

2. 다국어 번역기

도착 언어를 입력값으로 받아 어떤 언어로든 번역합니다. system으로 "번역가" 역할을 고정.

prompt = ChatPromptTemplate.from_messages([
    ("system", "너는 번역가야. 입력 문장을 {target}로만 자연스럽게 번역하고 다른 말은 하지 마."),
    ("human", "{text}"),
])
translator = prompt | llm | StrOutputParser()

# target 만 바꾸면 어떤 언어로든 번역
print(translator.invoke({"text": "좋은 아침이에요", "target": "English"}))
print(translator.invoke({"text": "좋은 아침이에요", "target": "일본어"}))

3. 감성·주제 분류기

문장을 정해진 라벨 중 하나로 분류합니다. "라벨 단어 하나만 출력"이 핵심 규칙.

labels = ["긍정", "부정", "중립"]      # 원하는 라벨로 자유롭게 바꾸기

prompt = ChatPromptTemplate.from_messages([
    ("system", f"다음 문장의 감정을 {labels} 중 하나로만 분류해. 라벨 단어 하나만 출력."),
    ("human", "{text}"),
])
classify = prompt | llm | StrOutputParser()

print(classify.invoke({"text": "배송도 빠르고 품질도 최고예요!"}))   # 긍정
print(classify.invoke({"text": "그냥 그래요. 무난합니다."}))         # 중립
💡 응용
labels를 ["배송","환불","상품문의","기타"]처럼 바꾸면 문의 유형 분류기가 됩니다.

4. 키워드·해시태그 추출 (JSON)

결과를 JSON으로 받으면 파이썬에서 바로 다룰 수 있습니다. JsonOutputParser가 문자열을 딕셔너리로 바꿔 줍니다.

from langchain_core.output_parsers import JsonOutputParser   # 답을 dict로 변환

prompt = ChatPromptTemplate.from_template(
    "다음 글의 핵심 키워드 5개와 해시태그 3개를 뽑아줘.\n"
    '반드시 JSON으로만: {{"keywords": [...], "hashtags": [...]}}\n\n{text}'
)
# JSON 출력엔 흔들림 0이 좋음 → 위 공통 llm(temperature=0) 사용
extract = prompt | llm | JsonOutputParser()

result = extract.invoke({"text": "제주 한라산 트레킹 코스와 준비물 안내..."})
print(result["keywords"])   # 파이썬 리스트로 바로 접근 가능
print(result["hashtags"])
📝 중괄호 주의
프롬프트 안에서 진짜 중괄호를 쓰려면 {{ }}처럼 두 번 씁니다(템플릿 변수 { }와 구분).

5. 이메일·공지 작성기

변수 여러 개를 받아 격식 있는 초안을 만듭니다. 받는사람·목적·톤만 바꿔 재사용.

prompt = ChatPromptTemplate.from_messages([
    ("system", "너는 비즈니스 글쓰기 도우미야. 정중하고 간결하게 한국어로 써."),
    ("human", "받는사람: {to}\n목적: {purpose}\n톤: {tone}\n위 내용으로 이메일 초안을 써줘."),
])
# 작성물은 약간의 자연스러움 → temperature 0.4 짜리 모델을 따로 지정
writer = prompt | ChatOpenAI(model="gpt-4o-mini", temperature=0.4) | StrOutputParser()

print(writer.invoke({
    "to": "교수님",
    "purpose": "과제 기한 3일 연장 요청",
    "tone": "아주 공손하게",
}))

6. 표(CSV) 데이터 Q&A

표 데이터를 글로 바꿔 프롬프트에 넣으면, 표에 대해 자연어로 물어볼 수 있습니다.

import pandas as pd

# 예시 표 (실제로는 pd.read_csv("파일.csv") 로 불러오기)
df = pd.DataFrame({
    "이름": ["김철수", "이영희", "박민수"],
    "점수": [88, 95, 72],
})

prompt = ChatPromptTemplate.from_template(
    "아래 표를 보고 질문에 한국어로 답해줘.\n\n[표]\n{table}\n\n[질문]\n{question}"
)
qa = prompt | llm | StrOutputParser()

# df.to_markdown(): 표를 모델이 읽기 좋은 글자표로 변환
print(qa.invoke({"table": df.to_markdown(index=False),
                 "question": "가장 점수가 높은 사람은 누구야?"}))
💡 큰 표는?
수천 행이면 통째로 넣지 말고, 먼저 pandas로 필요한 행·열만 추린 뒤 넣으세요(토큰=비용).

7. FAQ 챗봇 — 미니 RAG

PDF 없이도, 질문/답 목록만으로 작은 RAG(검색 후 답변)를 만들 수 있습니다. Day 4~5의 축소판.

from langchain_chroma import Chroma
from langchain_openai import OpenAIEmbeddings
from langchain_core.runnables import RunnablePassthrough

# 우리 서비스 FAQ (원하는 만큼 추가)
faqs = [
    "환불은 구매 후 7일 이내 가능합니다.",
    "배송은 보통 2~3일 걸립니다.",
    "회원 등급은 누적 구매액으로 자동 산정됩니다.",
]
# FAQ 문장들을 임베딩해 벡터DB로 → 질문과 가까운 1개를 찾는 검색기
db = Chroma.from_texts(faqs, OpenAIEmbeddings())
retriever = db.as_retriever(search_kwargs={"k": 1})

prompt = ChatPromptTemplate.from_template(
    "아래 FAQ만 근거로 한국어로 답하라. 없으면 '잘 모르겠어요'라고 답하라.\n[FAQ]\n{context}\n\n[질문]\n{question}"
)
def join_docs(docs):                       # 찾은 문서들의 본문을 한 덩어리로
    return "\n".join(d.page_content for d in docs)

rag = ({"context": retriever | join_docs, "question": RunnablePassthrough()}
       | prompt | llm | StrOutputParser())

print(rag.invoke("환불 며칠 안에 돼요?"))   # FAQ에서 근거를 찾아 답함

8. 유튜브 자막 요약

영상 자막을 받아 핵심만 요약합니다. 긴 영상도 3줄로.

!pip install youtube-transcript-api

from youtube_transcript_api import YouTubeTranscriptApi

video_id = "동영상ID"      # youtube.com/watch?v=여기 부분
# 한국어(ko) 우선, 없으면 영어(en) 자막 가져오기
parts = YouTubeTranscriptApi.get_transcript(video_id, languages=["ko", "en"])
script = " ".join(p["text"] for p in parts)   # 자막 조각들을 한 글로 합침

prompt = ChatPromptTemplate.from_template(
    "다음 영상 자막을 한국어 3줄로 요약하고, 키워드 5개를 덧붙여줘.\n\n{script}"
)
summarize = prompt | llm | StrOutputParser()
print(summarize.invoke({"script": script[:6000]}))   # 너무 길면 앞부분만(토큰 절약)

9. 멀티턴 기억 챗봇

앞의 대화를 기억하는 챗봇입니다. RunnableWithMessageHistory가 세션별 기록을 자동 관리합니다.

from langchain_core.prompts import MessagesPlaceholder
from langchain_core.chat_history import InMemoryChatMessageHistory
from langchain_core.runnables.history import RunnableWithMessageHistory

prompt = ChatPromptTemplate.from_messages([
    ("system", "너는 친절한 한국어 비서야."),
    MessagesPlaceholder(variable_name="history"),   # 과거 대화가 들어갈 자리
    ("human", "{input}"),
])
chain = prompt | llm | StrOutputParser()

store = {}                                   # 세션ID → 대화기록
def get_history(session_id):
    if session_id not in store:
        store[session_id] = InMemoryChatMessageHistory()
    return store[session_id]

bot = RunnableWithMessageHistory(
    chain, get_history,
    input_messages_key="input",      # 입력 키
    history_messages_key="history",  # 과거 대화 자리 이름(Placeholder와 일치)
)

cfg = {"configurable": {"session_id": "userA"}}   # 같은 세션이면 대화가 이어짐
print(bot.invoke({"input": "내 이름은 민수야."}, config=cfg))
print(bot.invoke({"input": "내 이름 기억해?"}, config=cfg))   # 민수를 기억함

10. 구조화 출력 (Pydantic)

정해진 형태(필드)로 안전하게 받고 싶을 때. with_structured_output이 결과를 객체로 채워 줍니다.

from pydantic import BaseModel, Field   # 데이터 형태를 정의하는 도구

# 받고 싶은 결과의 "틀"을 클래스로 정의
class 후기분석(BaseModel):
    감정: str = Field(description="긍정/부정/중립 중 하나")
    별점: int = Field(description="1~5 사이 정수")
    한줄평: str = Field(description="핵심을 담은 한 문장")

# 모델에게 이 틀대로 답하라고 지정 → 결과가 후기분석 객체로 옴
structured = llm.with_structured_output(후기분석)

r = structured.invoke("음식은 맛있는데 너무 오래 기다렸어요.")
print(r.감정, r.별점)   # 점 표기로 필드 접근
print(r.한줄평)
📝 왜 좋나
문자열을 직접 파싱할 필요 없이 r.별점 + 1처럼 바로 계산·저장할 수 있어 실수가 줄어듭니다.

11. 스트리밍 + Gradio 한 줄 배포

ChatGPT처럼 답이 실시간으로 흐르게 하고, 웹 화면까지 한 번에 붙입니다.

!pip install gradio
import gradio as gr

# 화면에서 메시지를 보낼 때마다 실행 — yield 로 부분 답을 계속 흘려보냄(스트리밍)
def respond(message, history):
    partial = ""
    for chunk in llm.stream(message):    # 답을 조각으로 받아
        partial += chunk.content         # 누적하며
        yield partial                    # 화면에 계속 갱신

# ChatInterface : 채팅 UI 한 줄. share=True → 외부 공유 URL 생성
gr.ChatInterface(respond, title="내 스트리밍 챗봇").launch(share=True)

12. 도구 호출 맛보기 (Tool calling)

모델이 필요할 때 내가 만든 함수를 직접 부르게 합니다. 계산·검색처럼 LLM이 약한 일을 함수에 맡기는 방식.

from langchain_core.tools import tool

# @tool 을 붙이면 일반 함수가 "모델이 부를 수 있는 도구"가 됨
@tool
def add(a: int, b: int) -> int:
    """두 정수를 더한다."""      # 설명(docstring) — 모델이 언제 쓸지 판단하는 근거
    return a + b

# 모델에게 도구 목록을 알려줌
llm_with_tools = llm.bind_tools([add])

# 모델이 "add(3, 5)를 부르자"고 판단하면 그 요청이 tool_calls 에 담겨 옴
msg = llm_with_tools.invoke("3 더하기 5는?")
print(msg.tool_calls)   # [{'name': 'add', 'args': {'a': 3, 'b': 5}, ...}]

# 실제 실행은 우리가: 요청된 인자로 함수를 호출
call = msg.tool_calls[0]
print(add.invoke(call["args"]))   # 8
💡 다음 단계
도구를 여러 개 묶고 결과를 다시 모델에 돌려주면 "스스로 도구를 골라 일하는" 에이전트(agent)가 됩니다.
